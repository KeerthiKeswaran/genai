import threading
import time

            self._pool.put(self._create_new_connection())

    def get_connection(self):
        with self._lock:  # Ensures thread safety when accessing the pool
            try:
                connection, timestamp = self._pool.get(timeout=self.connection_timeout)

                    raise psycopg2.InterfaceError("Connection has been stale")

            return connection, time.time()
        finally:
            with self._lock:  # Ensures thread safety when returning the connection to the pool
                if not psycopg2.connect().closed:
                    psycopg2.dispose(connection)
        
    def _create_new_connection(self):
        try:
            with self._lock:  # Ensures thread safety when creating a new connection
                conn = psycopg2.connect()
                return conn, time.time()
        except Exception as e:
            raise ConnectionError("Unable to create database connection") from e
