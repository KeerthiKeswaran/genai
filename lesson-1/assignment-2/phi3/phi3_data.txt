```sql
SELECT c.customer_id, COUNT(DISTINCT oi.product_id) AS unique_products_bought, SUM(oi.quantity * oi.unit_price) AS total_spending
FROM orders AS o 
INNER JOIN customers AS c ON o.customer_id = c.customer_id
LEFT JOIN order_items AS oi ON o.order_id = oi.order_id AND DATE(o.order_date, '+30 days') <= CURRENT_DATE
GROUP BY c.customer_id
ORDER BY total_spending DESC
LIMIT 5;
```
